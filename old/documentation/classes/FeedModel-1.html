  type_media: string;

  get_middle_url_port: string;
  get_small_url_land: string;
  get_middle_url_land: string;

  get_middle_url_square: string;
}

export interface Feed {
  id: number;
  title: string;
  text: string;
  lon: number;
  lat: number;
  city: string;
  country: string;
  has_video: boolean;
  is_deleted: boolean;
  is_approved: boolean;
  videos: FeedMedia[];
  photos: FeedMedia[];
}

export interface FeedState {
  ids: number[];
  results: {[id: number]: any};
}

export const default_state = {
  ids:  [],
  results: {}
}


export class FeedModel implements Feed {
  id: number;
  title: string;
  text: string;
  lon: number;
  lat: number;
  city: string;
  country: string;
  has_video: boolean;
  is_deleted: boolean;
  is_approved: boolean;
  videos: FeedMedia[];
  photos: FeedMedia[];

  constructor(feed?)
  {
      feed = feed || {};
      this.id = feed.id || null;
      this.title = feed.title || '';
      this.text = feed.text || '';
      this.videos = feed.videos || [];
      this.photos = feed.photos || [];

  }


}
