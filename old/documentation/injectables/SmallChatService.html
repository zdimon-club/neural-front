import { SoundService } from './../services/sound.service';

import { SocketService } from './../../socket/socket.service';
import { Injectable } from '@angular/core';
import { BehaviorSubject, Observable, Subject } from 'rxjs';
import { Router, NavigationStart, NavigationEnd, NavigationError, NavigationCancel, RoutesRecognized } from '@angular/router';


// store
import { selectCurrentRoom } from '../../main/chat/store/chat.selector';
import { Store } from '@ngrx/store';
import { Room, RoomState } from '../../main/chat/store/chat.store';

// snack bar alert
import { SnackbarService } from './../snackbar/snackbar.service';
import { SnackbarMessage } from './../snackbar/snackbar.model';

@Injectable()
export class SmallChatService {

  public chatState = new BehaviorSubject<boolean>(false);
  public chatDisabled = new BehaviorSubject<boolean>(false);

  private room: any;

  constructor(
    router:Router,
    private socket_service: SocketService,
    private room_store: Store<RoomState>,
    private alert: SnackbarService,
    private sound_service: SoundService
    ) {


    router.events.forEach((event) => {
      if(event instanceof NavigationEnd) {
          if(event.url.indexOf('main/chat') == -1)
          {
            this.chatState.next(true);
            this.chatDisabled.next(false);
          } else {
            this.chatDisabled.next(true);
            this.chatDisabled.next(true);
          }
          
      }
     
    });

    // new message outsidethe current room

    this.socket_service.chat_message$.subscribe(data => {
      console.log(data.data);
      if(this.room == undefined) {
        this.showAlert(data.data.data);
      } else if (this.room.id != data.data.data.id) {
        this.showAlert(data.data.data);
      }
      this.sound_service.playMessage();

    })


    // current chat room

    this.room_store.select(selectCurrentRoom)
    .subscribe(data =>{
      this.room = data;
    });

   }

  

  get chatOpened(): Observable<boolean> {
    return this.chatState.asObservable();
  }

    /// show snack bar

    showAlert(data: any){
      this.alert.showToast(data);
    }

}
