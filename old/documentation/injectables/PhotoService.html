import { Photo } from './store/photos.store';

import {Injectable, Inject} from '@angular/core';
import {HttpClient} from '@angular/common/http';
import { Injector } from '@angular/core';
import {ReplaySubject} from 'rxjs';
import { environment } from '../../../environments/environment';


@Injectable()
export class PhotoService {

  _camera_emmiter$ = new ReplaySubject();

  constructor(
    private http: HttpClient,
    public injector: Injector,

  ) {
  }

  public picTaken(){
      this._camera_emmiter$.next();
  }

  public getList(page:number, role: string) {
    console.log(role);
    page==1 ? page = 0: page = page;
    let offset = page*10;
    return this.http.get(`${environment.apiUrl}/usermedia/photo/list/${role}`);
  }

  public save(photo: Photo){
    return this.http.put(`${environment.apiUrl}/usermedia/photo/${photo.id}/`,photo);
  }

  public add(photo: Photo){
    return this.http.post(`${environment.apiUrl}/usermedia/photo`,photo);
  }

  public setMain(photo: Photo){
    return this.http.post(`${environment.apiUrl}/usermedia/setmain`,photo);
  }

  public delete(photo: Photo){
    return this.http.post(`${environment.apiUrl}/usermedia/delete`,photo);
  }

  public sendPhoto(data: any) {
    return this.http.post(`${environment.apiUrl}/usermedia/add/image`, data);
  }

  public sendCrop(data: any) {
    return this.http.post(`${environment.apiUrl}/usermedia/crop`, data);
  }

  public sendModerate(data: any) {
    return this.http.post(`${environment.apiUrl}/usermedia/photo`, data);
  }

  public changeRole(data: any) {
    return this.http.post(`${environment.apiUrl}/usermedia/change/role`, data);
  }
}
