import {Injectable, Inject} from '@angular/core';
import {HttpClient} from '@angular/common/http';
import {environment}  from '../../../../environments/environment';
import { SessionState } from '../../../auth/store/session.store';
import { getSessionStateSelector } from '../../../auth/store/session.selector';
import { Store } from '@ngrx/store';
import * as sessionActions from '../../../auth/store/session.action';

@Injectable()
export class WebRtcService {

  constructor(
    private http: HttpClient,
    private session_store: Store<SessionState>
    ){
      this.session_store.select(getSessionStateSelector);
    }

  public init() {
    console.log('Init webrtc');
  }

  public sendOffer(data: any){
    return this.http.post(`${environment.apiUrl}/webrtc/offer`, data );
  }

  public sendAnswer(data: any){
    return this.http.post(`${environment.apiUrl}/webrtc/answer`, data );
  }

  public sendIce(data: any){
    return this.http.post(`${environment.apiUrl}/webrtc/ice`, data );
  } 

  public cameraOn(){
    return this.http.get(`${environment.apiUrl}/webrtc/cameraOn` );
  } 

  public cameraOff(){
    return this.http.get(`${environment.apiUrl}/webrtc/cameraOff` );
  }

  public cameraShow(data: any){
    return this.http.post(`${environment.apiUrl}/webrtc/cameraShow`, data);
  }

  public cameraHide(data: any){
    return this.http.post(`${environment.apiUrl}/webrtc/cameraHide`, data);
  }

}
