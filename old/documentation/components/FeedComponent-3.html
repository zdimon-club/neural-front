  public addComment(id, comment): void {
    if (id > 0 && comment){
      const data = {
        id: id,
        comment: comment,
      };
      this.feedService.addComment(data).subscribe((res => {
        this.openSnackBar(res.message, 'ok');
      }));
    }
  }

  openSnackBar(message: string, action: string): void {
    this._snackBar.open(message, action, {
      duration: 2000,
    });
  }
