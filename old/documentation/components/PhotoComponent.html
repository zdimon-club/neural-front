import { Component, OnInit, ViewEncapsulation, Output, EventEmitter, ViewChild, Inject, OnDestroy } from '@angular/core';
import { Subscription, Observable } from 'rxjs';
import { Subject } from 'rxjs';
import { takeUntil } from 'rxjs/operators';
import { ActivatedRoute, Router } from '@angular/router';

/// Store
import { Store } from '@ngrx/store';
import { PhotoState, Photo } from './store/photos.store';
import { getPhotoList } from './store/photos.selector';
import * as PhotoActions from './store/photos.action';
import * as sessionActions from '../../auth/store/session.action';
///////

// services
import { HttpClient } from '@angular/common/http';
import { SessionService } from './../../auth/session.service';
import { SocketService } from './../../socket/socket.service';
import { PhotoService } from './photo.service';
import { MatSnackBar } from '@angular/material/snack-bar';

import { fuseAnimations } from '@fuse/animations';

// dialog
import {MatDialog, MatDialogRef, MAT_DIALOG_DATA} from '@angular/material/dialog';
import { CameraDialogComponent } from './camera.component';



@Component({
    selector     : 'app-photo',
    templateUrl  : './photo.component.html',
    styleUrls    : ['./photo.component.scss'],
    encapsulation: ViewEncapsulation.None,
    animations   : fuseAnimations
})
export class PhotoComponent implements OnInit, OnDestroy
{

    // Private
    private _unsubscribeAll: Subject<any>;
    private files: any;

    role: string = 'all';
    editphoto: Photo;


    displayedColumns = [  
      "get_small_url_square", 
      "role_media", 
      "is_main", 
      "is_deleted", 
      "is_approved", 
      "edit", 
      "delete"];

    photos: Observable<any>;
    currentPage: number = 1;
    pageType: string;


    constructor(
        private store: Store<PhotoState>,
        private http_service: PhotoService,
        private http: HttpClient,
        private session_service: SessionService,
        private socket_service: SocketService,
        public dialog: MatDialog,
        private photo_service: PhotoService,
        private router: Router,
        private _matSnackBar: MatSnackBar,
    )
    {
        this._unsubscribeAll = new Subject();
        this.photos = this.store.select(getPhotoList);
        /*
        this.store.select(getPhotoList).subscribe((data: any) => {
            this.photos = data;
           });
        */
        this.photo_service._camera_emmiter$.subscribe(data => {
            this.getPage(this.currentPage);
        });

        this.socket_service.photo_moderation$
           .pipe(takeUntil(this._unsubscribeAll))
           .subscribe(data => {
             this.getPage(this.currentPage);
           })

    }


    ngOnInit(): void
    {
        this.getPage(this.currentPage);
    }

    openDialog(): void {
        const dialogRef = this.dialog.open(CameraDialogComponent, {
          width: '450px',
          data: {name: 'dima'}
        });
    }


    roleChanged(event: any, photo: any){
      //console.log(event.value);
      let data = {
        value: event.value,
        photo: photo
      }
      this.photo_service.changeRole(data).subscribe(rez => {
        this._matSnackBar.open(rez['message'], 'OK', {
          verticalPosition: 'top',
          duration        : 2000
        });
      })
    }

    getPage(page: number){
        //console.log(`Getting page ${page}`);
        this.currentPage = page;
        this.store.dispatch(new PhotoActions.Get({page: this.currentPage, role: this.role}));
    }



      photoUploaded(event){
        this.getPage(this.currentPage);
      }

      doSetMain(photo: Photo){
        photo.is_main = true;
        //this.store.dispatch(new PhotoActions.SetMain(photo));
        this.store.dispatch(new sessionActions.SetMainPhoto(photo.get_small_url_square));
        this.photo_service.setMain(photo).subscribe((data => {
           this.getPage(this.currentPage);
        }))
      }

      doDelete(photo: Photo){

        photo.is_deleted = true;
        //this.store.dispatch(new PhotoActions.Delete(photo));
        this.photo_service.delete(photo).subscribe((data)=>{
          this.getPage(this.currentPage);
        })
      }

      doEditPhoto(photo){
        this.editphoto = photo;
      }

      onFileChanged(event) {
        this.files = event.target.files[0];
        this.onUpload();
      }

      onUpload() {
        const formData = new FormData();
        formData.append('myfile', this.files, this.files.name);
        this.photo_service.sendPhoto(formData)
          .subscribe((rez: any) => {
            
            this._matSnackBar.open(rez.message, 'OK', {
                verticalPosition: 'top',
                duration        : 2000
              });
            if(rez.status == 0) {
              this.store.dispatch(new PhotoActions.Add(rez));
              this.router.navigate(['/main/photo/edit',rez[Object.keys(rez)[0]].id]);
            } 
          });
      }

    ngOnDestroy(){
        // Unsubscribe from all subscriptions
        this._unsubscribeAll.next();
        this._unsubscribeAll.complete();
    }
}
