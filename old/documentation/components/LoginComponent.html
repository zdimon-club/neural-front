import { Component, DoCheck, OnInit, ViewEncapsulation } from '@angular/core';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { LoginService } from '../../auth/login.service';

// auth google module
import { AuthService } from 'angularx-social-login';
import { GoogleLoginProvider } from 'angularx-social-login';

@Component({
  selector: 'app-login',
  templateUrl: './login.component.html',
  styleUrls: ['./login.component.scss']
})
export class LoginComponent implements OnInit, DoCheck {
  loginForm: FormGroup;

  /**
   * Constructor
   *
   * @param {FuseConfigService} _fuseConfigService
   * @param {FormBuilder} _formBuilder
   */
  is_error: boolean;
  error_message: string;


  public tempUsers = {
    men: [],
    women: []
  };
  public allUsers: any;

  constructor(
    private _formBuilder: FormBuilder,
    private _loginService: LoginService,
    private authService: AuthService
  ) {
    for (let i = 1; i <= 10; i++) {
      const man = {
        name: `Man${i}`,
        value: {username: `man${i}`, password: `man${i}`}
      };
      const woman = {
        name: `Woman${i}`,
        value: {username: `woman${i}`, password: `woman${i}`}
      };
      this.tempUsers.men.push(man);
      this.tempUsers.women.push(woman);
    }
    this.allUsers = this.tempUsers;

  }

  // -----------------------------------------------------------------------------------------------------
  // @ Lifecycle hooks
  // -----------------------------------------------------------------------------------------------------

  /**
   * On init
   */
  ngOnInit(): void {
    this.loginForm = this._formBuilder.group({
      // email   : ['', [Validators.required, Validators.email]],
      username: ['', [Validators.required, Validators.email]],
      password: ['', Validators.required]
    });
    this._loginService.login$.subscribe((data: any) => {
      this.is_error = true;
      this.error_message = data.message;
    });

    this.authService.authState.subscribe((user) => {
     // this._loginService.loginGoogle(user);
    });

  }

  ngDoCheck(): void {

  }

  onSubmit() {
    this._loginService.login(this.loginForm.value);
  }

  public loginAs(user) {
    this._loginService.login(user);
  }

  signInWithGoogle(): void {
    // console.log('ssss');
    this.authService.signIn(GoogleLoginProvider.PROVIDER_ID);
    this.authService.authState.subscribe(user => {
      this._loginService.loginGoogle(user);
    });
  }
}
