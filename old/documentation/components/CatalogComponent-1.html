import { CatalogService } from './catalog.service';
import { Component, OnInit, OnDestroy, ViewEncapsulation } from '@angular/core';
import { Store } from '@ngrx/store';
import { Subject } from 'rxjs';
import { takeUntil } from 'rxjs/operators';

// store 
import { User, UserState } from '../users/store/users.store';
import { selectUserList, selectGalleryUserList } from '../users/store/users.selector';
import { UpdateUsers } from '../users/store/users.action';
import { SessionState } from '../../auth/store/session.store';
import { selectUser } from '../../auth/store/session.selector';
import { Router, ActivatedRoute } from '@angular/router';


import { MatSnackBar } from '@angular/material/snack-bar';

// services
import { ChatService } from '../chat/services/chat.service';

@Component({
  selector: 'app-catalog',
  templateUrl: './catalog.component.html',
  styleUrls: ['./catalog.component.scss'],
  encapsulation: ViewEncapsulation.None,
  
})
export class CatalogComponent implements OnInit, OnDestroy {

  private _unsubscribeAll: Subject<any>;

  users: any;
  current_user: any;
  filter: any = {
    'online': 'false',
    'age_from': 'all',
    'age_to': 'all'
  };

  constructor(
    private user_store: Store<UserState>,
    private catalog_service: CatalogService,
    private session_store: Store<SessionState>,
    private router: Router,
    private route: ActivatedRoute,
    private chat_service: ChatService,
    private _matSnackBar: MatSnackBar,
  ) { 
    this._unsubscribeAll = new Subject();
  }

  ngOnInit() {

    let online = this.route.snapshot.paramMap.get('isOnline');
    if (online == 'true') this.filter.online = 'true'

    this.session_store.select(selectUser)
    .pipe(takeUntil(this._unsubscribeAll))
    .subscribe(user => {
      this.current_user = user;
    });
    this.doGetList(this.filter);
  }

  doGetList(filter: any){
    this.catalog_service.getList(filter).subscribe((res:any) => {
      this.user_store.dispatch(new UpdateUsers(res.results));
      this.users = this.user_store.select(selectGalleryUserList(res.results));
    })
  }

  callToChat(user_id: number) {
    let data = {
      'owner': this.current_user.id,
      'abonent': user_id
    };
    this.chat_service.addRoom(data).subscribe(data => {
      this.router.navigate(['/main/chat/room', data['id']]);
    });
  }

  public subscribe(user){
    this.catalog_service.subscribe(user.id)
    .subscribe((data: any) => {
      console.log(data);
      user.is_subscribed = true;
      this._matSnackBar.open(data.message, 'OK', {
        verticalPosition: 'top',
        duration        : 2000
      });
    })
  }

  doFilter(event: any){
    this.doGetList(event);
    this.router.navigate(["/main/catalog/online", event.online]);
  }

  ngOnDestroy(){
    // Unsubscribe from all subscriptions
    this._unsubscribeAll.next();
    this._unsubscribeAll.complete();
  }


}
